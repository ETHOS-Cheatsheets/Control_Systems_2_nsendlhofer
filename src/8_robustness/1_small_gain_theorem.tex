        \begin{itemize}
            \item $G(s)$ is a known, stable system
            \item $\Delta(s)$ is unknown, assumed to be stable and $\doubleabs{\Delta}_{\mathcal{H}_\infty} < 1$
            \item Interconnection is stable iff $\doubleabs{G}_{\mathcal{H}_\infty} < 1$
        \end{itemize}
