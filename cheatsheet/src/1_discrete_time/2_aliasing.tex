    \begin{align*}
        y_1[k] &= cos(\omega k T_s), &&k = 0, 1, 2\\
        y_2[k] &= cos((\omega + n\frac{2\pi}{T_s})k T_s), &&n = 0, 1, 2\\
        &= cos(\omega k T_s + \text{\cancel{$n2\pi k$}}) = y_1[k]
    \end{align*}
